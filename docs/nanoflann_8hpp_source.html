159template <
160 typename _DistanceType, typename _IndexType = size_t,
161 typename _CountType = size_t>
162class KNNResultSet
163{
164 public:
165 using DistanceType = _DistanceType;
166 using IndexType = _IndexType;
167 using CountType = _CountType;
168
169 private:
170 IndexType* indices;
171 DistanceType* dists;
172 CountType capacity;
173 CountType count;
174
175 public:
176 explicit KNNResultSet(CountType capacity_)
177 : indices(nullptr), dists(nullptr), capacity(capacity_), count(0)
178 {
179 }
180
181 void init(IndexType* indices_, DistanceType* dists_)
182 {
183 indices = indices_;
184 dists = dists_;
185 count = 0;
186 if (capacity)
187 dists[capacity - 1] = (std::numeric_limits<DistanceType>::max)();
188 }
189
190 CountType size() const { return count; }
191
192 bool full() const { return count == capacity; }
193
199 bool addPoint(DistanceType dist, IndexType index)
200 {
201 CountType i;
202 for (i = count; i > 0; --i)
203 {
206#ifdef NANOFLANN_FIRST_MATCH
207 if ((dists[i - 1] > dist) ||
208 ((dist == dists[i - 1]) && (indices[i - 1] > index)))
209 {
210#else
211 if (dists[i - 1] > dist)
212 {
213#endif
214 if (i < capacity)
215 {
216 dists[i] = dists[i - 1];
217 indices[i] = indices[i - 1];
218 }
219 }
220 else
221 break;
222 }
223 if (i < capacity)
224 {
225 dists[i] = dist;
226 indices[i] = index;
227 }
228 if (count < capacity) count++;
229
230 // tell caller that the search shall continue
231 return true;
232 }
233
234 DistanceType worstDist() const { return dists[capacity - 1]; }
235};

380template <
381 class T, class DataSource, typename _DistanceType = T,
382 typename IndexType = uint32_t>
383struct L1_Adaptor
384{
385 using ElementType = T;
386 using DistanceType = _DistanceType;
387
388 const DataSource& data_source;
389
390 L1_Adaptor(const DataSource& _data_source) : data_source(_data_source) {}
391
392 DistanceType evalMetric(
393 const T* a, const IndexType b_idx, size_t size,
394 DistanceType worst_dist = -1) const
395 {
396 DistanceType result = DistanceType();
397 const T* last = a + size;
398 const T* lastgroup = last - 3;
399 size_t d = 0;
400
401 /* Process 4 items with each loop for efficiency. */
402 while (a < lastgroup)
403 {
404 const DistanceType diff0 =
405 std::abs(a[0] - data_source.kdtree_get_pt(b_idx, d++));
406 const DistanceType diff1 =
407 std::abs(a[1] - data_source.kdtree_get_pt(b_idx, d++));
408 const DistanceType diff2 =
409 std::abs(a[2] - data_source.kdtree_get_pt(b_idx, d++));
410 const DistanceType diff3 =
411 std::abs(a[3] - data_source.kdtree_get_pt(b_idx, d++));
412 result += diff0 + diff1 + diff2 + diff3;
413 a += 4;
414 if ((worst_dist > 0) && (result > worst_dist)) { return result; }
415 }
416 /* Process last 0-3 components. Not needed for standard vector lengths.
417 */
418 while (a < last)
419 { result += std::abs(*a++ - data_source.kdtree_get_pt(b_idx, d++)); }
420 return result;
421 }
422
423 template <typename U, typename V>
424 DistanceType accum_dist(const U a, const V b, const size_t) const
425 {
426 return std::abs(a - b);
427 }

736class PooledAllocator
737{
738 static constexpr size_t WORDSIZE = 16;
739 static constexpr size_t BLOCKSIZE = 8192;
740
741 /* We maintain memory alignment to word boundaries by requiring that all
742 allocations be in multiples of the machine wordsize. */
743 /* Size of machine word in bytes. Must be power of 2. */
744 /* Minimum number of bytes requested at a time from the system. Must be
745 * multiple of WORDSIZE. */
746
747 using Offset = uint32_t;
748 using Size = uint32_t;
749 using Dimension = int32_t;
750
751 Size remaining_ = 0;
752 void* base_ = nullptr;
753 void* loc_ = nullptr;
754
755 void internal_init()
756 {
757 remaining_ = 0;
758 base_ = nullptr;
759 usedMemory = 0;
760 wastedMemory = 0;
761 }
762
763 public:
764 Size usedMemory = 0;
765 Size wastedMemory = 0;
766
770 PooledAllocator() { internal_init(); }
771
775 ~PooledAllocator() { free_all(); }
776
778 void free_all()
779 {
780 while (base_ != nullptr)
781 {
782 // Get pointer to prev block
783 void* prev = *(static_cast<void**>(base_));
784 ::free(base_);
785 base_ = prev;
786 }
787 internal_init();
788 }
789
794 void* malloc(const size_t req_size)
795 {
796 /* Round size up to a multiple of wordsize. The following expression
797 only works for WORDSIZE that is a power of 2, by masking last bits
798 of incremented size to zero.
799 */
800 const Size size = (req_size + (WORDSIZE - 1)) & ~(WORDSIZE - 1);
801
802 /* Check whether a new block must be allocated. Note that the first
803 word of a block is reserved for a pointer to the previous block.
804 */
805 if (size > remaining_)
806 {
807 wastedMemory += remaining_;
808
809 /* Allocate new storage. */
810 const Size blocksize =
811 (size + sizeof(void*) + (WORDSIZE - 1) > BLOCKSIZE)
812 ? size + sizeof(void*) + (WORDSIZE - 1)
813 : BLOCKSIZE;
814
815 // use the standard C malloc to allocate memory
816 void* m = ::malloc(blocksize);
817 if (!m)
818 {
819 fprintf(stderr, "Failed to allocate memory.\n");
820 throw std::bad_alloc();
821 }
822
823 /* Fill first word of new block with pointer to previous block. */
824 static_cast<void**>(m)[0] = base_;
825 base_ = m;
826
827 Size shift = 0;
828 // int size_t = (WORDSIZE - ( (((size_t)m) + sizeof(void*)) &
829 // (WORDSIZE-1))) & (WORDSIZE-1);
830
831 remaining_ = blocksize - sizeof(void*) - shift;
832 loc_ = (static_cast<char*>(m) + sizeof(void*) + shift);
833 }
834 void* rloc = loc_;
835 loc_ = static_cast<char*>(loc_) + size;
836 remaining_ -= size;
837
838 usedMemory += size;
839
840 return rloc;
841 }

1028 NodePtr divideTree(
1029 Derived& obj, const Offset left, const Offset right, BoundingBox& bbox)
1030 {
1031 NodePtr node = obj.pool_.template allocate<Node>(); // allocate memory
1032 const auto dims = (DIM > 0 ? DIM : obj.dim_);
1033
1034 /* If too few exemplars remain, then make this a leaf node. */
1035 if ((right - left) <= static_cast<Offset>(obj.leaf_max_size_))
1036 {
1037 node->child1 = node->child2 = nullptr; /* Mark as leaf node. */
1038 node->node_type.lr.left = left;
1039 node->node_type.lr.right = right;
1040
1041 // compute bounding-box of leaf points
1042 for (Dimension i = 0; i < dims; ++i)
1043 {
1044 bbox[i].low = dataset_get(obj, obj.vAcc_[left], i);
1045 bbox[i].high = dataset_get(obj, obj.vAcc_[left], i);
1046 }
1047 for (Offset k = left + 1; k < right; ++k)
1048 {
1049 for (Dimension i = 0; i < dims; ++i)
1050 {
1051 const auto val = dataset_get(obj, obj.vAcc_[k], i);
1052 if (bbox[i].low > val) bbox[i].low = val;
1053 if (bbox[i].high < val) bbox[i].high = val;
1054 }
1055 }
1056 }
1057 else
1058 {
1059 Offset idx;
1060 Dimension cutfeat;
1061 DistanceType cutval;
1062 middleSplit_(obj, left, right - left, idx, cutfeat, cutval, bbox);
1063
1064 node->node_type.sub.divfeat = cutfeat;
1065
1066 BoundingBox left_bbox(bbox);
1067 left_bbox[cutfeat].high = cutval;
1068 node->child1 = this->divideTree(obj, left, left + idx, left_bbox);
1069
1070 BoundingBox right_bbox(bbox);
1071 right_bbox[cutfeat].low = cutval;
1072 node->child2 = this->divideTree(obj, left + idx, right, right_bbox);
1073
1074 node->node_type.sub.divlow = left_bbox[cutfeat].high;
1075 node->node_type.sub.divhigh = right_bbox[cutfeat].low;
1076
1077 for (Dimension i = 0; i < dims; ++i)
1078 {
1079 bbox[i].low = std::min(left_bbox[i].low, right_bbox[i].low);
1080 bbox[i].high = std::max(left_bbox[i].high, right_bbox[i].high);
1081 }
1082 }
1083
1084 return node;
1085 }

1087 void middleSplit_(
1088 const Derived& obj, const Offset ind, const Size count, Offset& index,
1089 Dimension& cutfeat, DistanceType& cutval, const BoundingBox& bbox)
1090 {
1091 const auto dims = (DIM > 0 ? DIM : obj.dim_);
1092 const auto EPS = static_cast<DistanceType>(0.00001);
1093 ElementType max_span = bbox[0].high - bbox[0].low;
1094 for (Dimension i = 1; i < dims; ++i)
1095 {
1096 ElementType span = bbox[i].high - bbox[i].low;
1097 if (span > max_span) { max_span = span; }
1098 }
1099 ElementType max_spread = -1;
1100 cutfeat = 0;
1101 for (Dimension i = 0; i < dims; ++i)
1102 {
1103 ElementType span = bbox[i].high - bbox[i].low;
1104 if (span > (1 - EPS) * max_span)
1105 {
1106 ElementType min_elem, max_elem;
1107 computeMinMax(obj, ind, count, i, min_elem, max_elem);
1108 ElementType spread = max_elem - min_elem;
1109 if (spread > max_spread)
1110 {
1111 cutfeat = i;
1112 max_spread = spread;
1113 }
1114 }
1115 }
1116 // split in the middle
1117 DistanceType split_val = (bbox[cutfeat].low + bbox[cutfeat].high) / 2;
1118 ElementType min_elem, max_elem;
1119 computeMinMax(obj, ind, count, cutfeat, min_elem, max_elem);
1120
1121 if (split_val < min_elem)
1122 cutval = min_elem;
1123 else if (split_val > max_elem)
1124 cutval = max_elem;
1125 else
1126 cutval = split_val;
1127
1128 Offset lim1, lim2;
1129 planeSplit(obj, ind, count, cutfeat, cutval, lim1, lim2);
1130
1131 if (lim1 > count / 2)
1132 index = lim1;
1133 else if (lim2 < count / 2)
1134 index = lim2;
1135 else
1136 index = count / 2;
1137 }

1148 void planeSplit(
1149 const Derived& obj, const Offset ind, const Size count,
1150 const Dimension cutfeat, const DistanceType& cutval, Offset& lim1,
1151 Offset& lim2)
1152 {
1153 /* Move vector indices for left subtree to front of list. */
1154 Offset left = 0;
1155 Offset right = count - 1;
1156 for (;;)
1157 {
1158 while (left <= right &&
1159 dataset_get(obj, vAcc_[ind + left], cutfeat) < cutval)
1160 ++left;
1161 while (right && left <= right &&
1162 dataset_get(obj, vAcc_[ind + right], cutfeat) >= cutval)
1163 --right;
1164 if (left > right || !right)
1165 break; // "!right" was added to support unsigned Index types
1166 std::swap(vAcc_[ind + left], vAcc_[ind + right]);
1167 ++left;
1168 --right;
1169 }
1170 /* If either list is empty, it means that all remaining features
1171 * are identical. Split in the middle to maintain a balanced tree.
1172 */
1173 lim1 = left;
1174 right = count - 1;
1175 for (;;)
1176 {
1177 while (left <= right &&
1178 dataset_get(obj, vAcc_[ind + left], cutfeat) <= cutval)
1179 ++left;
1180 while (right && left <= right &&
1181 dataset_get(obj, vAcc_[ind + right], cutfeat) > cutval)
1182 --right;
1183 if (left > right || !right)
1184 break; // "!right" was added to support unsigned Index types
1185 std::swap(vAcc_[ind + left], vAcc_[ind + right]);
1186 ++left;
1187 --right;
1188 }
1189 lim2 = left;
1190 }

1594 template <class RESULTSET>
1595 bool searchLevel(
1596 RESULTSET& result_set, const ElementType* vec, const NodePtr node,
1597 DistanceType mindist, distance_vector_t& dists,
1598 const float epsError) const
1599 {
1600 /* If this is a leaf node, then do check and return. */
1601 if ((node->child1 == nullptr) && (node->child2 == nullptr))
1602 {
1603 DistanceType worst_dist = result_set.worstDist();
1604 for (Offset i = node->node_type.lr.left;
1605 i < node->node_type.lr.right; ++i)
1606 {
1607 const IndexType accessor = Base::vAcc_[i]; // reorder... : i;
1608 DistanceType dist = distance_.evalMetric(
1609 vec, accessor, (DIM > 0 ? DIM : Base::dim_));
1610 if (dist < worst_dist)
1611 {
1612 if (!result_set.addPoint(dist, Base::vAcc_[i]))
1613 {
1614 // the resultset doesn't want to receive any more
1615 // points, we're done searching!
1616 return false;
1617 }
1618 }
1619 }
1620 return true;
1621 }
1622
1623 /* Which child branch should be taken first? */
1624 Dimension idx = node->node_type.sub.divfeat;
1625 ElementType val = vec[idx];
1626 DistanceType diff1 = val - node->node_type.sub.divlow;
1627 DistanceType diff2 = val - node->node_type.sub.divhigh;
1628
1629 NodePtr bestChild;
1630 NodePtr otherChild;
1631 DistanceType cut_dist;
1632 if ((diff1 + diff2) < 0)
1633 {
1634 bestChild = node->child1;
1635 otherChild = node->child2;
1636 cut_dist =
1637 distance_.accum_dist(val, node->node_type.sub.divhigh, idx);
1638 }
1639 else
1640 {
1641 bestChild = node->child2;
1642 otherChild = node->child1;
1643 cut_dist =
1644 distance_.accum_dist(val, node->node_type.sub.divlow, idx);
1645 }
1646
1647 /* Call recursively to search next level down. */
1648 if (!searchLevel(
1649 result_set, vec, bestChild, mindist, dists, epsError))
1650 {
1651 // the resultset doesn't want to receive any more points, we're done
1652 // searching!
1653 return false;
1654 }
1655
1656 DistanceType dst = dists[idx];
1657 mindist = mindist + cut_dist - dst;
1658 dists[idx] = cut_dist;
1659 if (mindist * epsError <= result_set.worstDist())
1660 {
1661 if (!searchLevel(
1662 result_set, vec, otherChild, mindist, dists, epsError))
1663 {
1664 // the resultset doesn't want to receive any more points, we're
1665 // done searching!
1666 return false;
1667 }
1668 }
1669 dists[idx] = dst;
1670 return true;
1671 }
